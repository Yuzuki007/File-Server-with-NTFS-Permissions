Configuration

NTFS Permission Setup

PowerShell — Set NTFS Permissions
# Reusable function to set clean NTFS permissions
function Set-FolderPermission {
  param($FolderPath, $GroupName, $Permission)

  $acl = Get-Acl $FolderPath

  # Step 1: Disable inheritance — stop parent permissions flowing down
  $acl.SetAccessRuleProtection($true, $false)

  # Step 2: Remove ALL existing permissions — clean slate
  $acl.Access | ForEach-Object { $acl.RemoveAccessRule($_) }

  # Step 3: Add Domain Admins — Full Control (IT admin access)
  $adminRule = New-Object System.Security.AccessControl.FileSystemAccessRule(
    "MYLAB\Domain Admins", "FullControl",
    "ContainerInherit,ObjectInherit", "None", "Allow")
  $acl.AddAccessRule($adminRule)

  # Step 4: Add department group with specified permission
  $groupRule = New-Object System.Security.AccessControl.FileSystemAccessRule(
    "MYLAB\$GroupName", $Permission,
    "ContainerInherit,ObjectInherit", "None", "Allow")
  $acl.AddAccessRule($groupRule)

  # Step 5: Apply the ACL to the folder
  Set-Acl $FolderPath $acl
}

# Apply to each department folder
Set-FolderPermission -FolderPath "C:\FileServer\IT"         -GroupName "IT-Team"         -Permission "Modify"
